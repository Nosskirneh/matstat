\section{Estimation}
\begin{math}
  \bar{X} = \frac{1}{n} \sum_{i = 1}^n X_i,
  S^2 = \frac{1}{n - 1} \sum_{i=1}^n (X_i - \bar{X})^2
\end{math} \\[2pt]
\begin{math}
  \hat{\theta} \text{ is unbiased if } \expect{\hat{\theta}} = \theta
\end{math} \\[2pt]
\begin{math}
  \widehat{p_1 - p_2} = \widehat{p_1} - \widehat{p_2} = \frac{X_1}{n} - \frac{X_2}{n}
\end{math} \\[2pt]

\subsection{Central Limit Theorem}
\begin{math}
  \text{If } |\expect{X}| < \infty \text{ and } Var(X) < \infty \text{ then}
\end{math} \\[2pt]
\begin{math}
  \bar{X} \sim N(\mu, \frac{\sigma^2}{n}) \Rightarrow \frac{\bar{X} - \mu}{\frac{\sigma^2}{n}} \sim N(0, 1)
\end{math} \\[2pt]

\subsection{Confidence Interval}
\begin{math}
  Z \sim N(0, 1), T \sim t(n - 1), C \sim \chi^2(n - 1)
\end{math} \\[2pt]
\begin{math}
  \mu \rightarrow
 \bar{X} \pm \cidist{Z} \cdot \frac{\sigma}{\sqrt{n}}
\end{math} \\[2pt]
\begin{math}
  \mu \rightarrow
 \bar{X} \pm \cidist{T} \cdot \frac{S}{\sqrt{n}}
\end{math} \\[2pt]
\begin{math}
  \sigma^2 \rightarrow
  \left[ \frac{(n - 1)S^2}{C_{\frac{\alpha}{2}}}, \frac{(n - 1)S^2}{C_{1 - \frac{\alpha}{2}}} \right]
\end{math} \\[2pt]
\begin{math}
  p \rightarrow
  \hat{p} \pm \cidist{Z} \cdot \sqrt{\frac{p(1 - p)}{n}}
\end{math} \\[2pt]
\begin{math}
  p_1 - p_2 \rightarrow
  \pm \cidist{Z} \sqrt{\frac{p_1(1 - p_1)}{n_1} + \frac{p_2(1 - p_2)}{n_2}}
\end{math} \\[2pt]
For one sided confidence intervals, use \(Z_{\alpha}\)
