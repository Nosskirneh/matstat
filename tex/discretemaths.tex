\section{Discrete Mathematics}
\begin{math}
  \binom{n}{k} = \frac{n!}{k!(n - k)!},
  \binom{n}{k} = \binom{n}{n - k} \text{ for } 0 \leq k \leq n
\end{math} \\[2pt]
Recusrive formula:
\begin{math}
  \binom{n}{k} = \binom{n - 1}{k - 1} + \binom{n - 1}{k}
\end{math} \\[2pt]
Pascal's rule:
\begin{math}
  \binom{n}{k} + \binom{n}{k + 1} = \binom{n + 1}{k + 1}
\end{math} \\[2pt]
