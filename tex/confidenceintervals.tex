\section{Confidence Intervals}
\begin{math}
  X_1, \dotsc, X_n; X_i \sim N(\mu, \sigma^2)
\end{math} \\[2pt]
\begin{math}
  \text{Use the CLT to get } \frac{\bar{X} - \mu}{\sigma/\sqrt{n}} \sim N(0,1)
\end{math} \\[2pt]
We can then construct a \(1 - \alpha\) level CI interval:
\begin{math}
  P(-Z_{\frac{\alpha}{2}} \leq \frac{\bar{X} - \mu}{\sigma / \sqrt{n}} \leq Z_{\frac{\alpha}{2}}) =
  P(\left| \frac{\bar{X} - \mu}{\sigma / \sqrt{n}} \right| \leq Z_{\frac{\alpha}{2}}) =
  P(|\bar{X} - \mu| \le Z_{\frac{\alpha}{2}} \cdot \frac{\sigma}{\sqrt{n}}) =
  P(\bar{X} - Z_{\frac{\alpha}{2}} \cdot \frac{\sigma}{\sqrt{n}} \le \mu \le \bar{X} + Z_{\frac{\alpha}{2}} \cdot \frac{\sigma}{\sqrt{n}}) =
  1 - \alpha
\end{math} \\[2pt]
